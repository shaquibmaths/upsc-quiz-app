        function checkAnswer(selectedElement, questionId, selectedKey) {

            const questionData = window.questionsData.find(q => q.id === questionId);

            if (!questionData) return;


            const correctKey = questionData.answer;

            const optionsContainer = selectedElement.parentElement;

            

            // Prevent further clicks if already answered

            if (optionsContainer.hasAttribute('data-answered')) return;

            optionsContainer.setAttribute('data-answered', 'true');


            const allOptions = optionsContainer.children;

            const explanationContainer = optionsContainer.nextElementSibling;


            for (const option of allOptions) {

                option.onclick = null; // Disable click

                option.classList.remove('hover:bg-gray-200', 'cursor-pointer');

                option.classList.add('cursor-not-allowed');


                const optionKey = option.textContent.substring(0, 1);

                if (optionKey === correctKey) {

                    option.classList.remove('bg-gray-50');

                    option.classList.add('bg-green-100', 'border-2', 'border-green-500', 'font-semibold');

                }

            }


            if (selectedKey !== correctKey) {

                selectedElement.classList.remove('bg-gray-50');

                selectedElement.classList.add('bg-red-100', 'border-2', 'border-red-500');

            }

            

            explanationContainer.classList.remove('hidden');

            setTimeout(() => explanationContainer.classList.remove('opacity-0'), 10);

        }
